func 피보나치수_재귀(n: Int) -> Int {
        guard n >= 1 else { return -1 }
        
        if n == 1 || n == 2 {
            return 1
        }
        
        return 피보나치수_재귀(n: n - 1) + 피보나치수_재귀(n: n - 2)
}
    
func 피보나치수_DP(n: Int) -> Int {
        var arr = [1,1]
        
        if n == 1 {
            return arr[0]
        }
        
        if n == 2 {
            return arr[1]
        }
        
        for i in 2 ... (n - 1) {
            let sum = arr[i - 1] + arr[i - 2]
            arr.append(sum)
        }
        
        return arr.last!
}
