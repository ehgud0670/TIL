func 튜플(_ s:String) -> [Int] {
        var arrs = [[Int]]()
        var arr = [Int]()
        var numStr = ""
        
        for char in s {
            // 숫자가 아닌 경우 { } ,
            if char == "," {
                guard let num = Int(numStr) else { continue }
                
                arr.append(num)
                numStr = ""
                continue
            }
            
            if char == "}" {
                guard let num = Int(numStr) else { continue }
                
                arr.append(num)
                numStr = ""
                arrs.append(arr)
                arr = [Int]()
                continue
            }
            
            // 숫자인 경우
            if Int(numStr) != nil {
                numStr.append(char)
                continue
            }
            numStr = String(char)
        }
        
        arrs.sort {$0.count < $1.count }
        var answer = [arrs[0].first!]
        
        guard arrs.count > 1 else { return answer }
        for i in 0 ..< arrs.count - 1 {
            
            let prev = arrs[i]
            var next = arrs[i + 1]
            next.removeAll {
                prev.contains($0)
            }
            
            answer.append(next.first!)
        }
        return answer
    }
